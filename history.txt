    file <- "discrete_model/matrices/q_matrix_" + i + ".csv"
    contents <- readDelimitedDataFile(file, header = TRUE, separator=",", rownames=FALSE)
    for (j in 1:contents.size()) {
        rates[i][j] <- contents[j][1]
    }
    n = rates[i].size()
    k = round((1 + sqrt(1 + 4 * n)) / 2)

}
moves = VectorMoves()
taxa <- morpho.names()
num_taxa <- taxa.size()
num_branches <- 2 * num_taxa - 2
br_len_lambda ~ dnExp(0.2)
moves.append( mvScale(br_len_lambda, weight=2) )
phylogeny ~ dnUniformTopologyBranchLength(taxa, branchLengthDistribution=dnExponential(br_len_lambda))
moves.append( mvNNI(phylogeny, weight=num_branches/2.0) )
moves.append( mvSPR(phylogeny, weight=num_branches/10.0) )
moves.append( mvBranchLengthScale(phylogeny, weight=num_branches) )
# compute the tree length from the phylogeny
tree_length := phylogeny.treeLength()
# Set up Gamma-distributed rate variation.
alpha_morpho ~ dnUniform( 0.0, 1E6 )
rates_morpho := fnDiscretizeGamma( alpha_morpho, alpha_morpho, 4 )
# Moves on the parameters to the Gamma distribution.
moves.append( mvScale(alpha_morpho,lambda=1, weight=2.0) )
for (i in 1:nchar) {
    print(i)
    morpho_part[i] <- morpho
    morpho_part[i].excludeAll()
    morpho_part[i].includeCharacter(i)

                               nSites=1,
                               siteRates=rates_morpho,
